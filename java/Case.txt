public class Case 
{ 
    public static void main(String []args){ 
           int mes= 3; 
           switch (mes){ 
           case 1: 
               System.out.println("Enero"); 
               break; 
           case 2: 
               System.out,println("Febrero"); 
               break;
           case 3: 
               System.out,println("Marzo"); 
               break; 
           default: 
               System.out,println("Mes no soportado"); 
               break; 
           } 
         } 
 }
